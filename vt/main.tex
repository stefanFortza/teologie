    \hline
    Biblia 1796 Blaj (Septuaginta)               & Și au zis Isaac către Avraam, tatăl său: „Tată”. Iară el au zis: „Ce easte, fiule?” Și au zis: „Iată focul și leamnele, unde easte oaia pentru jertvă?”                                   \\
    \hline
    Biblia 1914 (Septuaginta)                    & Și a zis Isaac către Avraam, tatăl său: tată, iar el a zis: ce este, fiule? Și a zis: iată focul și lemnele, unde este oaea pentru jertfă?                                                \\
    \hline
    Biblia 2001 Anania (Septuaginta)             & Dar Isaac i-a grăit lui Avraam, tatăl său; i-a zis: „Tată!” Iar acesta a răspuns: „Ce este, fiule?” [Isaac] a zis: „Iată focul şi lemnele; dar mielul pentru jertfă unde este?”           \\
    \hline
    Septuaginta I (Septuaginta)                  & Isaac i-a zis tatălui său, Avraam: „Tată.” Iar el l-a întrebat: „Ce-i, copile?” Celălalt: „Avem focul și lemnele, dar unde este oaia pentru jertfă?”                                      \\

\section{Comentarii Patristice}

\subsection{Sfântul Ioan Gură de Aur: Comentariu la versetul 22:5}

\begin{quote}
    \textit{„Și le-a zis Avraam slugilor sale: Voi rămâneți aici, cu asinul; eu și băiețelul ne ducem pân'acolo, ne vom închina și ne vom întoarce la voi."}
\end{quote}

Pentru acest verset, Sfântul Ioan Gură de Aur comentează:

\begin{quote}
    „Uită-te și aici la marea pricepere a dreptului Avraam! Voia să se ascundă și de slugile sale, ca să arate prin toate că dorește să împlinească porunca lui Dumnezeu cu râvnă clocotitoare și dor înflăcărat. Pentru că știa că fapta ce trebuia s-o facă era nouă și neobișnuită și că nimeni altul nu făcuse mai înainte ce avea să facă el."
\end{quote}

\vspace{0.5cm}
\noindent\textbf{Sursa:} Sfântul Ioan Gură de Aur, \emph{Omilii la Facere}, Omilia 47, capitolul 2.



\subsection{Origen: Comentariu la versetul 2}

\begin{quote}
    \textit{„Mergi în pământ înalt, pe
        unul din munți, pe care Eu ți-l voi arăta și acolo
        oferă-l pe el ardere de tot!"}
\end{quote}

Pentru acest verset Origen comentează:
\begin{quote}
    „...mai întâi i se
    spune că trebuie să-l ofere pe fiul său și abia apoi se
    poruncește să meargă \emph{în pământ înalt} și să urce pe
    munte. Pentru ce aceasta?

    Pentru că atât timp cât merge, cât se află pe
    drum, să fie sfâșiat de gânduri, să fie torturat cât e
    calea de lungă: porunca îl hărțuia pe de o parte, iar
    pe de altă parte îl tortura iubirea față de unicul fiu."
\end{quote}

\vspace{0.5cm}
\noindent\textbf{Sursa:} Origen, \emph{Omilii la Geneză}, Omilia 8, Paragraf 3.

\section{Comentarii Moderne}
\subsection{William MacDonald: Comentariu la versetul 7}

\begin{quote}
    \textit{„Dar Isaac i-a grăit lui Avraam, tatăl său; i-a zis: „Tată!” Iar acesta a
    răspuns: „Ce este, fiule?” [Isaac] a zis:
    „Iată focul şi lemnele; dar mielul pentru jertfă unde este?"}
\end{quote}

Pentru acest verset William MacDonald comentează:
\begin{quote}
    „Când Isaac l-a întrebat <<\textbf{Unde este mielul}?>>, tatăl lui i-a
    răspuns, <<\textbf{Dumnezeu Însuși va purta de grijă de mielul pentru arderea de tot}>>. Această promisiune nu a fost împlinită definitiv de berbecul din versetul 13, ci de Mielul lui Dumnezeu (Ioan 1:29).
    "
\end{quote}

\vspace{0.5cm}
\noindent\textbf{Sursa:} William MacDonald, \emph{Cometariul biblic al credinciosului. Vechiul Testament}, Editura Lampadarul de Aur, Oradea, 2002, p. 56

\subsection{Vasile Tarnavschi: Comentariu la versetul 8}

\begin{quote}
    \textit{„Iar Avraam a răspuns: „Dumnezeu
        va vedea El Însuşi de mielul pentru
        jertfă, fiule!” Şi s’au dus amândoi mai
        departe, împreună.
        "}
\end{quote}

Pentru acest verset Vasile Tarnavschi comentează:
\begin{quote}
    „Răspunsul lui Abraam se poate privi ca o încunjurare a întrebării fiului său, căci inima sa părintească nu-i permite să răspundă: tu vei fi olocaustul;
    acest răspuns însă poate să aibă şi un sens mai profund: Abraam sperează
    cumcă Dumnezeu îl va elibera din aceasta dilemă de tot greu.
    "
\end{quote}

\vspace{0.5cm}
\noindent\textbf{Sursa:} Vasile Tarnavschi, \emph{Geneza
    Introducere Și Comentar}, Editura Societatea tipografica bucovineană
, Cernăuți, 1907, p. 161
